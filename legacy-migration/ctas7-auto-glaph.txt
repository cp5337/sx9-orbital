js// Auto-triggers on edit
function onEdit(e) {
  const sheet = e.source.getActiveSheet();
  if (sheet.getName() === "Ground Stations") {
    const data = sheet.getDataRange().getValues();
    fetch("http://glaf.ctas7.mil/api/ingest/sheet", {
      method: "POST",
      body: JSON.stringify({ data, sheetId: e.source.getId() })
    });
  }
}
